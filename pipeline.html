Pipeline NGINX-LAB

node {
        stage('Deletando arquivos antigos') {
                 sh "rm -rf /home/ubuntu/jenkins" 

         }
       stage('Fazendo download do projeto') {
                sh "git clone https://github.com/Extars01/Case.git /home/ubuntu/jenkins"

        }
       stage('Construindo imagem da aplicação') {
                sh "sudo docker build -t image-nginx /home/ubuntu/jenkins"

        }
       stage('Derrumando versão antiga da aplicação') {
                sh "sudo docker stop site && docker rm site" 

         }
       stage('Levantando aplicação atualizada'){
                sh "sudo docker run --name site -p 80:80 -d image-nginx" 
 
        }
}
